// Simple generative bloom effect

float angle = 0;
float petalCount = 8;
float bloomScale = 0;
float maxBloomScale = 200;
float bloomSpeed = 0.5;

void setup() {
  size(800, 800);
  background(0);
  noStroke();
}

void draw() {
  // Fading background for continuous motion
  fill(0, 5);
  rect(0, 0, width, height);

  translate(width / 2, height / 2); // Center drawing

  // Color changes, simulating sound
  float r = map(sin(frameCount * 0.01), -1, 1, 50, 255);
  float g = map(cos(frameCount * 0.015), -1, 1, 50, 255);
  float b = map(sin(frameCount * 0.005), -1, 1, 50, 255);
  fill(r, g, b, 150); // Semi-transparent petals

  // Update bloom size
  bloomScale += bloomSpeed;
  if (bloomScale > maxBloomScale) {
    bloomScale = 0; // Reset bloom
  }

  // Petals drawn
  for (int i = 0; i < petalCount; i++) {
    pushMatrix();
    rotate(angle + TWO_PI / petalCount * i);
    ellipse(0, bloomScale / 2, bloomScale / (petalCount / 2), bloomScale);
  }

  angle += 0.01; // Continuous rotation
}
                                    

// Dynamic cityscape reacting to mouse X

int numBuildings = 50;
float[] buildingHeights = new float[numBuildings];
float[] buildingWidths = new float[numBuildings];
float[] buildingX = new float[numBuildings];
float buildingBase = 700; // Ground level

void setup() {
  size(800, 800);
  for (int i = 0; i < numBuildings; i++) {
    buildingX[i] = random(width);
    buildingHeights[i] = random(50, 300);
    buildingWidths[i] = random(20, 80);
  }
  noStroke();
}

void draw() {
  background(20, 20, 40); // Dark sky

  // City activity based on mouseX
  float activity = map(mouseX, 0, width, 0.5, 1.5);

  for (int i = 0; i < numBuildings; i++) {
    // Height changes with activity
    float currentHeight = buildingHeights[i] * activity;

    // Color changes with activity
    fill(map(activity, 0.5, 1.5, 50, 200), map(activity, 0.5, 1.5, 80, 220), map(activity, 0.5, 1.5, 100, 255), 200);

    rect(buildingX[i], buildingBase - currentHeight, buildingWidths[i], currentHeight);

    // Random window lights
    if (random(1) > 0.8) {
      fill(255, 200, 50, 150);
      rect(buildingX[i] + random(5, buildingWidths[i] - 10), buildingBase - currentHeight + random(5, currentHeight - 10), 5, 5);
    }
  }
}
                                    

// Generates a simple plant with random visual glitches

void setup() {
  size(800, 800);
  background(0);
  stroke(255);
  noFill();
  strokeWeight(2);
}

void draw() {
  background(0, 10); // Subtle fade
  translate(width/2, height*0.9); // Plant base

  // Main stem
  stroke(100, 200, 50);
  line(0, 0, 0, -300);

  // Branches and leaves
  drawBranch(0, -300, 100, 1);

  // Random glitches added
  if (random(1) < 0.1) {
    drawGlitch();
  }
}

void drawBranch(float x, float y, float len, int level) {
  if (len < 10) return; // Stop drawing small branches

  pushMatrix();
  translate(x, y);

  // Branching logic
  pushMatrix();
  rotate(radians(random(-30, -10)));
  line(0, 0, 0, -len);
  drawBranch(0, -len, len * 0.7, level + 1);
  popMatrix();

  pushMatrix();
  rotate(radians(random(10, 30)));
  line(0, 0, 0, -len);
  drawBranch(0, -len, len * 0.7, level + 1);
  popMatrix();

  popMatrix();
}

void drawGlitch() {
  fill(random(255), random(255), random(255), 150); // Colorful glitch fill
  noStroke();
  rectMode(CENTER);
  float gx = random(width);
  float gy = random(height);
  float gw = random(10, 80);
  float gh = random(10, 80);
  rect(gx, gy, gw, gh);
}
                                    

// Fading trails from a moving object, like time-sculpting

PVector[] positions; // Stores past positions
int maxPositions = 100; // How many past positions to remember
int currentIndex = 0;

void setup() {
  size(800, 800);
  background(0);
  noStroke();
  positions = new PVector[maxPositions];
}

void draw() {
  // Background slowly fades for trail effect
  fill(0, 20);
  rect(0, 0, width, height);

  // Store current mouse position
  positions[currentIndex] = new PVector(mouseX, mouseY);
  currentIndex = (currentIndex + 1) % maxPositions;

  // Draw all stored positions, older ones fade
  for (int i = 0; i < maxPositions; i++) {
    int index = (currentIndex + i) % maxPositions;
    if (positions[index] != null) {
      float alpha = map(i, 0, maxPositions, 0, 200); // Fade transparent to opaque
      float size = map(i, 0, maxPositions, 5, 30);   // Size grows
      fill(255, alpha); // White trails
      ellipse(positions[index].x, positions[index].y, size, size);
    }
  }

  // Draw current object (mouse cursor)
  fill(255, 255, 0); // Bright yellow
  ellipse(mouseX, mouseY, 40, 40);
}
                                    

// Abstract patterns influenced by mouse position (simulated biofeedback)

float lineLength = 50;
float turbulence = 0.01; // Controls chaos

void setup() {
  size(800, 800);
  background(0);
  strokeWeight(1);
}

void draw() {
  // Subtle background fade
  fill(0, 5);
  rect(0, 0, width, height);

  // "Emotional state" from mouseX
  float emotionIntensity = map(mouseX, 0, width, 0.05, 0.5);
  float colorHue = map(mouseX, 0, width, 0, 255);

  colorMode(HSB, 255);
  stroke(colorHue, 200, 255, 100); // Semi-transparent, saturated lines

  // Draw many lines per frame for a weave effect
  for (int i = 0; i < 5; i++) {
    float x1 = random(width);
    float y1 = random(height);

    // Emotion to end point
    float x2 = x1 + sin(frameCount * turbulence + x1 * emotionIntensity) * lineLength;
    float y2 = y1 + cos(frameCount * turbulence + y1 * emotionIntensity) * lineLength;

    line(x1, y1, x2, y2);
  }
}
